获取对话的最后一条消息

Implementation 

Future<EMMessage> getLastMessage() async {
  Map<String, dynamic> result = await _emConversationChannel
      .invokeMethod(EMSDKMethod.getLastMessage, {"id": _conversationId});
  if (result['success']) {
    return EMMessage.from(result['message']);
  }
  return null;
}
